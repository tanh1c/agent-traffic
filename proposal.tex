\section{TỔNG QUAN DỰ ÁN}
% ============================================

\subsection{Tên dự án}
\textbf{TrafficFlow Agent} — Hệ thống Agentic AI Tự động Điều phối Giao thông Thông minh

\subsection{Elevator Pitch}

\textbf{Vấn đề:} Mỗi ngày, hàng triệu người dân TP.HCM mắc kẹt trong ùn tắc tại các ngã tư như Trường Chinh – Cách Mạng Tháng 8, lãng phí hàng giờ chờ đợi dưới nắng mưa. Hệ thống đèn tín hiệu chạy theo lịch cố định không thích ứng với thực tế, trong khi cảnh sát giao thông phải điều phối thủ công trong điều kiện khắc nghiệt.

\textbf{Giải pháp:} TrafficFlow Agent là hệ thống AI tự chủ sử dụng camera real-time và dữ liệu Google Maps để phát hiện ùn tắt theo phương pháp hybrid (mật độ + tốc độ + độ dài hàng đợi), suy luận qua LLM đa nhà cung cấp (Groq / OpenRouter / Alibaba Qwen) để đưa ra quyết định điều chỉnh đèn tín hiệu tối ưu.

\textbf{Đối tượng:} Người tham gia giao thông tại TP.HCM, đặc biệt khu vực đông đúc; cảnh sát giao thông (CSGT); cơ quan quản lý giao thông đô thị.

\textbf{Giá trị mang lại:} Giảm thời gian chờ trung bình 15–25\%, giảm độ dài hàng đợi 15–30\%, tăng throughput 10–20\%, giảm phụ thuộc vào điều phối thủ công của CSGT, giảm ô nhiễm môi trường do kẹt xe kéo dài.

\subsection{Tại sao Agentic AI?}

Hệ thống của bạn thể hiện tính Agentic như thế nào?

\begin{itemize}
    \item[\textbf{\textbullet}] \textbf{Goal:} Agent luôn có mục tiêu rõ ràng — giảm thời gian chờ trung bình và tránh gridlock. Mọi hành động đều được đo lường và đánh giá theo KPI cụ thể.
    \item[\textbf{\textbullet}] \textbf{Planning:} Agent dùng LLM (Groq/OpenRouter/Alibaba Qwen) để phân tích tình hình giao thông đa chiều (density, speed, queue, thời gian, ngày trong tuần), đề xuất chiến lược điều chỉnh đèn với reasoning chain rõ ràng — không chỉ là rule-based đơn giản.
    \item[\textbf{\textbullet}] \textbf{Autonomy:} Agent tự quyết định và tự điều chỉnh cho các tình huống thông thường mà không cần chờ lệnh từ con người. Chỉ hỏi human approval khi phát hiện tình huống nghiêm trọng (tai nạn, sự kiện đặc biệt).
    \item[\textbf{\textbullet}] \textbf{Feedback Loop:} Sau mỗi hành động điều chỉnh đèn, agent liên tục observe lại trạng thái giao thông mới qua camera và Google Maps $\rightarrow$ re-analyze $\rightarrow$ re-decide. Cycle liên tục để đảm bảo hệ thống luôn thích ứng với thực tế.
\end{itemize}

\section{BÀI TOÁN VÀ GIẢI PHÁP}
% ============================================

\subsection{Problem Statement}

\subsubsection{Ai bị ảnh hưởng?}
\begin{itemize}
    \item Hàng triệu người dân TP.HCM — đặc biệt khu vực Trường Chinh, Cách Mạng Tháng 8, quận Tân Bình, quận 3
    \item Tài xế xe buýt, tài xế xe công nghệ (Grab, Be, Gojek)
    \item Cảnh sát giao thông (CSGT) phải điều phối thủ công giờ cao điểm
    \item Doanh nghiệp logistics chịu thiệt hại do chậm trễ vận chuyển
\end{itemize}

\subsubsection{Tại sao quan trọng?}
\begin{itemize}
    \item TP.HCM có $>$ 8 triệu xe máy, $>$ 800.000 ô tô (theo số liệu Sở GTVT)
    \item Ùn tắt gây thiệt hại kinh tế ước tính 1.2–1.5\% GDP mỗi năm
    \item Khí thải từ xe chờ ùn tắt gây ô nhiễm, ảnh hưởng sức khỏe
    \item CSGT làm việc quá tải trong điều kiện thời tiết khắc nghiệt
\end{itemize}

\subsubsection{Giải pháp hiện tại còn thiếu gì?}
\begin{itemize}
    \item Đèn tín hiệu tại hầu hết các ngã tư chạy theo lịch cố định, không thích ứng với mật độ thực tế
    \item Không có hệ thống tự động phát hiện + phản ứng nhanh với ùn tắt theo real-time
    \item Google Maps chỉ thấy ``đoạn đường đỏ'' (GPS probe data) nhưng không biết nguyên nhân và không điều khiển được đèn
    \item CSGT phải có mặt trực tiếp hoặc được gọi đến khi tình huống xấu đi
\end{itemize}

\subsection{Key Questions}

\begin{enumerate}
    \item \textbf{Làm sao AI có thể tự giải quyết bài toán giao thông?} $\rightarrow$ AI phân tích dữ liệu từ camera (YOLO) + Google Maps theo thời gian thực, đánh giá mức độ ùn tắt, và đưa ra quyết định điều chỉnh đèn dựa trên reasoning — không chỉ rule cố định.

    \item \textbf{Làm sao AI dùng nhiều công cụ (tools)?} $\rightarrow$ Agent dùng đồng thời: (1) YOLO để detect và đếm xe, (2) Google Maps API để lấy speed data, (3) LLM để reason và đề xuất action, (4) Simulation engine để visualize kết quả, (5) Dashboard để human-in-the-loop khi cần.

    \item \textbf{Làm sao đảm bảo độ tin cậy của hệ thống?} $\rightarrow$ Human-in-the-loop: agent chỉ tự động apply cho case thường; case nghiêm trọng (tai nạn, gridlock) luôn chờ approval từ CSGT trước khi hành động. A/B test evaluation đo lường hiệu quả thực tế qua simulation. Multi-provider LLM fallback đảm bảo hệ thống không bị down khi một provider lỗi.
\end{enumerate}

\subsection{Proposed Solution}

\subsubsection{System Overview}

TrafficFlow Agent là hệ thống hybrid gồm 3 lớp dữ liệu:

\begin{enumerate}
    \item \textbf{Camera Real-time (YOLO):} Chi tiết cục bộ — số xe từng làn, loại phương tiện, queue length cụ thể
    \item \textbf{Google Maps Traffic API:} Tầm nhìn tổng thể — speed per segment, congestion color, incident reports
    \item \textbf{AI Agent (LLM):} Congestion Detection $\rightarrow$ Reasoning \& Planning $\rightarrow$ Decision Engine
\end{enumerate}

Đầu ra:
\begin{itemize}
    \item Simulation Engine (Demo)
    \item SCATS/ITMS API (Future-ready)
    \item Dashboard CSGT (Real-time monitoring + Human approval)
\end{itemize}

\subsubsection{Agent Workflow}

\begin{enumerate}
    \item \textbf{Input:} Camera frames (YOLO) + Google Maps data $\rightarrow$ ghép thành bức tranh toàn diện
    \item \textbf{Analysis:} Congestion Detector phân tích density + speed + queue length $\rightarrow$ phân loại mức độ ùn tắt
    \item \textbf{Planning:} LLM đa nhà cung cấp suy luận tình huống, đề xuất chiến lược điều chỉnh đèn
    \item \textbf{Action:} Apply quyết định vào simulation (demo) hoặc SCATS/ITMS (thực tế)
    \item \textbf{Feedback:} Quan sát kết quả $\rightarrow$ loop lại bước 1
\end{enumerate}

\subsubsection{Key Features}

\begin{enumerate}
    \item \textbf{Hybrid Congestion Detection:} Kết hợp camera (YOLO) + Google Maps để phát hiện ùn tắt chính xác hơn bất kỳ giải pháp đơn lẻ nào. Camera thấy chi tiết từng làn; Maps thấy bức tranh toàn tuyến.

    \item \textbf{LLM-Powered Reasoning Agent:} Agent dùng LLM (Groq / OpenRouter / Alibaba Qwen) để reason theo chuỗi, đề xuất action cụ thể kèm giải thích. Multi-provider fallback đảm bảo độ sẵn sàng.

    \item \textbf{Human-in-the-Loop Dashboard:} CSGT giám sát real-time trên dashboard, nhận notification khi agent cần approval cho case nghiêm trọng. Quyết định được apply sau 30s nếu không ai phản hồi.
\end{enumerate}

\subsubsection{Unique Selling Point (USP)}

\begin{table}[h]
    \centering
    \begin{tabular}{|l|p{6cm}|p{6cm}|}
        \hline
        \textbf{So sánh} & \textbf{Google Maps Traffic} & \textbf{TrafficFlow Agent} \\
        \hline
        Dữ liệu & GPS probe (điện thoại ẩn danh) & Camera real-time + GPS \\
        \hline
        Độ chi tiết & Toàn tuyến, không phân biệt làn & Từng làn, từng loại xe \\
        \hline
        Phát hiện sự cố & Không (chỉ thấy đỏ) & Phát hiện tai nạn, vật cản \\
        \hline
        Điều khiển đèn & Không & Có (qua SCATS/ITMS API) \\
        \hline
        Tính Agentic & Không & Có (reasoning, planning, autonomy) \\

    \end{tabular}
\end{table}

\textbf{Điểm khác biệt cốt lõi:} Camera thấy được ``tại sao'' đoạn đường đỏ (tai nạn? làn nào tắt? xe gì?) trong khi Google Maps chỉ thấy ``đoạn đường đỏ'' mà không biết nguyên nhân.

\section{STACK CÔNG NGHỆ}
% ============================================

\subsection{AI \& Machine Learning}

\begin{itemize}
    \item \textbf{LLM:} Groq API / OpenRouter / Alibaba Qwen (multi-provider, flexible — dùng nhiều nguồn để đảm bảo availability + tối ưu chi phí)
    \item \textbf{Agent Framework:} LangChain / LangGraph (để xây reasoning chain rõ ràng, trace được)
    \item \textbf{Object Detection:} YOLO v8 (vehicle counting, classification, tracking) — xử lý real-time trên edge hoặc cloud
    \item \textbf{Embedding:} Không cần (dùng LLM trực tiếp cho reasoning)
\end{itemize}

\subsection{Infrastructure}

\begin{itemize}
    \item \textbf{Cloud:} Google Cloud Platform (align với GDG — có thể dùng Google Cloud credits từ hackathon)
    \item \textbf{Backend:} Next.js API routes (frontend + backend cùng stack, đơn giản hóa deployment)
    \item \textbf{Deployment:} Vercel / Google Cloud Run (fast, free tier đủ cho demo)
\end{itemize}

\subsection{Development}

\begin{itemize}
    \item \textbf{Frontend:} Next.js 14+ (App Router) + React + Tailwind CSS
    \item \textbf{Database:} Firebase Firestore (real-time state cho dashboard)
    \item \textbf{Simulation Engine:} Custom web-based (Canvas/SVG cho visualization)
    \item \textbf{Camera Processing:} YOLO v8 (via Python backend hoặc ONNX runtime on browser)
\end{itemize}

\subsection{Competitive Advantage}

\textbf{Điểm mạnh công nghệ:}
\begin{itemize}
    \item Kết hợp 3 nguồn dữ liệu: Camera + Maps + Weather (contextual)
    \item Multi-provider LLM: không phụ thuộc một nhà cung cấp duy nhất
    \item YOLO real-time detection: xử lý video stream trực tiếp, không cần gửi raw video
\end{itemize}

\textbf{Khả năng mở rộng:}
\begin{itemize}
    \item Thêm ngã tư mới: chỉ cần thêm camera + update config
    \item Mở rộng features: ưu tiên xe cấp cứu, dự báo tắt, tích hợp thêm Maps API
    \item Scale lên toàn thành phố: dùng distributed agent coordination
\end{itemize}

\textbf{Tính độc đáo:}
\begin{itemize}
    \item Chưa có giải pháp tại Việt Nam kết hợp camera + LLM agent + Maps cho điều phối đèn tín hiệu
    \item Human-in-the-loop design an toàn và thực tế cho context Việt Nam
\end{itemize}
